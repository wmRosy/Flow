;1. If 0xC2[2] = 1
;
;  	For DOVDD = 2.8V
;		0xB8[6:3] should be 4'b0001
;	For DOVDD = 1.8V
;		0xB8[6:3] should be 4'b0010
;
;   If 0xC2[2] = 0
;
;  	For DOVDD = 2.8V
;		0xDA[3:0] should be 4'b0001
;	For DOVDD = 1.8V
;		0xDA[3:0] should be 4'b0010
;
;2. Register 0xC1[6:0] control the accesibility of part of the 
;   registers. 
;	
;	Each bits control the respective registers block,
;
;	        1'b1 = Read/Write
;		1'b0 = Read only
;
;	0xc1[6]: Block register 0x1A
;	0xc1[5]: Block register 0x41, 0x75 to 0x77 (edge and de-noise )
;	0xc1[4]: Block register 0x24 to 0x26 (AEC)
;	0xc1[3]: Block register 0x62 to 0x66, 0x94, 0x95 (LENC)
;	0xc1[2]: Block register 0x6F (AWB)
;	0xc1[1]: Block register 0x4F to 0x54, 0x58 (color matrix)
;	0xc1[0]: Block register 0x7A to 0x89 (GAMMA)
;
